CREATE TABLE STUD1(REGNO NUMBER(6) PRIMARY KEY,MARK1 NUMBER(3),MARK2 NUMBER(3),MARK3 NUMBER(3),MARK4 NUMBER(3),MARK5 NUMBER(3),TOT NUMBER(3),AVG NUMBER(3,2));
DESC STUD1
INSERT INTO STUD1(REGNO,MARK1,MARK2,MARK3,MARK4,MARK5)VALUES(:REGNO,:MARK1,:MARK2,:MARK3,:MARK4,:MARK5);
SELECT * FROM STUD1;

DECLARE
CURSOR  C IS SELECT * FROM STUD1;
X STUD1.REGNO%TYPE;
A STUD1.MARK1%TYPE;
B STUD1.MARK2%TYPE;
F STUD1.MARK3%TYPE;
D STUD1.MARK4%TYPE;
E STUD1.MARK5%TYPE;
T STUD1.TOT%TYPE;
V STUD1.AVG%TYPE;
BEGIN
OPEN C;
LOOP
FETCH C INTO X,A,B,F,D,E,T,V;
EXIT WHEN C% NOTFOUND;
DBMS_OUTPUT.PUT_LINE('ROLLNO: '||X);
DBMS_OUTPUT.PUT_LINE('MARK1: '||A);
DBMS_OUTPUT.PUT_LINE('MARK2: '||B);
DBMS_OUTPUT.PUT_LINE('MARK3: '||F);
DBMS_OUTPUT.PUT_LINE('MARK4: '||D);
DBMS_OUTPUT.PUT_LINE('MARK5: '||E);
DBMS_OUTPUT.PUT_LINE('UPDATE THE TOT');
DBMS_OUTPUT.PUT_LINE('UPDATE THE AVG');
UPDATE STUD1 SET
TOT=(MARK1+MARK2+MARK3+MARK4+MARK5) WHERE REGNO=X;
UPDATE STUD1 SET AVG=(T/5) WHERE REGNO=X;
END LOOP;
CLOSE C;
END;
