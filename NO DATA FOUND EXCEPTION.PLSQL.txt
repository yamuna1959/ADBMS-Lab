CREATE TABLE EMP5(EMPNO NUMBER(3),ENAME VARCHAR(10),DEPTNO NUMBER(3));
INSERT INTO EMP5 VALUES(:EMPNO,:ENAME,:DEPTNO);
SELECT * FROM EMP5;

DECLARE
VENO EMP5.EMPNO%TYPE:=:EMPNO;
VENAME EMP5.ENAME%TYPE:=:ENAME;
VDEPTNO EMP5.DEPTNO%TYPE;
BEGIN
SELECT EMPNO,ENAME,DEPTNO INTO VENO,VENAME,VDEPTNO FROM EMP5 WHERE EMPNO=VENO AND ENAME=VENAME;
DBMS_OUTPUT.PUT_LINE('DEPTNO IS '||VDEPTNO);
EXCEPTION
WHEN NO_DATA_FOUND THEN
DBMS_OUTPUT.PUT_LINE('NO RECORDS FOUND');
END;
