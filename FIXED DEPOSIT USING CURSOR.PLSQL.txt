CREATE TABLE FIXED(ACCNO NUMBER(10),YEARS NUMBER(4),AMOUNT NUMBER(6),INTEREST NUMBER(3));
DESC FIXED
INSERT INTO FIXED(ACCNO,YEARS,AMOUNT) VALUES(:ACCNO,:YEARS,:AMOUNT);
SELECT * FROM FIXED;
DECLARE
CURSOR CUR IS SELECT * FROM FIXED;
BEGIN
FOR I IN CUR
LOOP
IF I.YEARS<=2 THEN
UPDATE FIXED SET INTEREST=I.AMOUNT*0.09*(1/12) WHERE ACCNO=I.ACCNO;
ELSIF I.YEARS>=2 AND I.YEARS<=5 THEN
UPDATE FIXED SET INTEREST=I.AMOUNT*0.11*(1/12) WHERE ACCNO=I.ACCNO;
ELSE
UPDATE FIXED SET INTEREST=I.AMOUNT*0.15*(1/12) WHERE ACCNO=I.ACCNO;
END IF;
END LOOP;
END;

SELECT * FROM FIXED;
